import java.util.ArrayDeque;
import java.util.Deque;
import java.util.Scanner;

//Construct queue using stack
public class Q2 {

    Deque<Integer> stack = new ArrayDeque<>();
    Deque<Integer> helper = new ArrayDeque<>();

    public void equeue(int data){
        stack.push(data);
        System.out.println("Data is equeue..");
    }

    public void shiftDataToHelper(){
        //stack to helperstack
        while(!stack.isEmpty()) {
            int temp = stack.pop();
            helper.push(temp);
        }
    }

    public void shiftDataToStack(){
        //move elm from helper to stack
        while(!helper.isEmpty()){
            int c = helper.pop();
            stack.push(c);
        }
    }

    public int dequeue(){

        shiftDataToHelper();
        int temp = helper.pop();
        shiftDataToStack();
        System.out.println("Data is dequeue " + temp);
        return temp;
    }

    public int peek(){
        //stack to helperstack
        shiftDataToHelper();
        int temp = helper.peek();
        shiftDataToStack();
        System.out.println("Peek is " + temp);
        return temp;
    }

    public boolean isEmpty(){
        return stack.isEmpty();
    }

    public void print(){
        shiftDataToHelper();
        System.out.println(helper);
        shiftDataToStack();
    }
